function dobro(x) {
    return x * 2
}

function triplo(x) {
    return x * 3
}

function executar(operacao, x) {
   if(operacao==="dobro"){
        return  dobro(x)
    }
    if(operacao==="triplo"){
        return triplo(x)
    }
}

// Exemplos de uso
console.log(executar("dobro", 5))
console.log(executar("triplo", 8))
